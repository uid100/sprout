Compare Values

Open IDLE and create a new file. Save it with a meaningful name. Copy the following code into your file, save it and run it.

Notice that (num_val1 > num_val2) always evaluates to true or false, regardless of what the numbers are. The expression will never equal a number!

#!/usr/bin/env python3

# compare numbers and strings demo

# Declare variables
num_val1 = 0
num_val2 = 0
str_val1 = ""
str_val2 = ""

# Let's compare numbers!
# get numeric input
num_val1 = int(input("Enter first number: "))
num_val2 = int(input("Enter second number: "))

# compare numbers
if num_val1 == num_val2:
    # values are equal
    print("values are equal")
elif num_val1 < num_val2:
    # val1 < val2
    print("value1 < value2")
else:
    # value1 > value2
    print("value1 > value2")

# Let's compare strings!
# get 2 strings for input
str_val1 = input("Enter first string: ")
str_val2 = input("Enter second string: ")

# compare strings - convert both strings to either all upper() or lower() case
# or use casefold() to do a case insensitive comparison
if str_val1.casefold() == str_val2.casefold():
    # values are equal
    print("string values are equal")
else:
    # value1 > value2
    print("String values are not equal")

Payroll Demo

 

#!/usr/bin/env python3

# add 2 numbers demo

# import locale to do currency formatting
import locale
locale.setlocale(locale.LC_ALL, 'en-US')


# Declare variables and constants
HOURS_IN_WEEK = 40
OVERTIME_RATE = 1.5
DENTAL_PREMIUM = 35.00
YES = 'Y'
hours_worked = 0.0
pay_rate = 0.0
pay = 0.0
deduct_dental = "N"


# get input
hours_worked = float(input("Enter hours worked: "))
pay_rate = float(input("Enter pay rate: "))
deduct_dental = input("Deduct dental premium (Y/N): ")
print("deduct_dental = " + deduct_dental)

# process - calculate gross pay
# indent 4 spaces for code in the true and false blocks
if hours_worked <= 40:
# normal work week
pay = hours_worked * pay_rate
else:
# calculate with overtime
pay = (pay_rate * HOURS_IN_WEEK) + ((hours_worked - HOURS_IN_WEEK) * (OVERTIME_RATE * pay_rate))

# Real world stuff the book leaves out: Social Security, Fed & state tax, etc
# check for dental deduction the casefold() method does case insensitive comparison
if deduct_dental.upper() == "Y":
pay = pay - DENTAL_PREMIUM


# output - display pay
print("Total = " + locale.currency(pay, grouping=True))

Phone bill using logical AND operator

 

#!/usr/bin/env python3

# logical AND demo

# import locale to do currency formatting
import locale
locale.setlocale(locale.LC_ALL, 'en-US') # use "en_US" on Mac

# Declare variables and constants
CALLS = 100
MINUTES = 500
BASIC_SERVICE = 30.00
ADD_PREMIUM = 20.00

customer_id = ""
calls_made = 0
call_minutes = 0
customer_bill = BASIC_SERVICE # start with basic service and accumulate from there...

# input - get customer ID, calls, minutes
customer_id = input("Enter Customer ID #: ")
calls_made = int(input("Enter number of calls: "))
call_minutes = int(input("Enter number of minutes: "))

# process - calculate bill using the AND operator. Both conditions must be true
if calls_made > CALLS and call_minutes > MINUTES:
    customer_bill = customer_bill + ADD_PREMIUM


# output - display customer bill
print("Total = " + locale.currency(customer_bill, grouping=True))

#!/usr/bin/env python3

# nested if demo

# import locale to do currency formatting
import locale

MINUTES = 500

ADD_PREMIUM = 20.00

Range demo (upper and lower limits)

 

#!/usr/bin/env python3

# Various coding techniques to calculate a discount based on qty
# Pardon the magic numbers

# Declare variables and constants
qty = 0
discunt = 0.0


# input - get qty
qty = int(input("Enter Qty: "))


# process - figure out the discount
# This approach uses upper and lower limits using the "and" operator
if qty >= 0 and qty <= 10:
    discount = 0.00
elif qty >= 11 and qty <= 24:
    discount = 0.10
elif qty >= 25 and qty <= 50:
    discount = 0.15
else:
    discount = 0.20


# output - display discount percentage
print("Discount is: " + "{:.2%}".format(discount))
