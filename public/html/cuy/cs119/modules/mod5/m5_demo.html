Array Demo

Using true arrays in Python

We will be using lists for most of our work, but Python does support arrays. Walk through this demo to see the difference.

We will need to import the array module for this

#!/usr/bin/env python3

# Cuyamaca College CS-119
# Array demo
# In Python, we need to import the array module
# If you get errors, execute the import statement in the Python shell.
import array as arr

# variables and constants
MAX_SIZE = 10

# declare an integer array of 4 elements

# nums = arr.array('i', [0,0,0,0])
print ("nums array")
nums = arr.array('i', range(4))

# initialize the array using sequential code
# Note the first subscript (index) starts at 0
nums[0] = 1
nums[1] = 2
nums[2] = 3
nums[3] = 4


# display the array. Note the for each loop
for num in nums:
    print(num, end=' ')
print("\n")
print ("nums2 array")

# A little bit bigger array and a loop to initialize the array elements
nums2 = arr.array('i', range(MAX_SIZE))
i = 0
for num in nums2:
    nums2[i] = i + 1
    i += 1

# display the bigger array
for num in nums2:
    print(num, end=' ')

print("\n")
print ("nums3 array")

# declare an array using literal constant values. This time we'll use float
nums3 = arr.array('d', [1.5, 3.1, 6.75, 8.0, 9.3, 7.6, 4.4])

# display the float array
for num in nums3:
    print(num, end=' ')

print("\n")
print ("nums4 array")

# Use the len() method to get the size of an array
nums4 = arr.array('i', range(MAX_SIZE))
i = 0
array_len = len(nums4)
for count in range(0, array_len, 1):
    nums4[count] = i + 1
    i += 1

# display the bigger array
for num in nums4:
    print(num, end=' ')

Array Search

Array search using Lists instead of Arrays

Lists are preferred in Python

#!/usr/bin/env python3

# Cuyamaca College CS-119
# In this demo we transition from arrays to lists as these are preferred
# in Python programming

# create a list of random animals and declare variables
animals = ["cat", "dog", "horse", "goat", "rabbit", "lizard", "duck"]
search_val = ""
i = 0
found = False
message = ""

# input - find out what we're looking for
search_val = input("Find what animal? ")

# process - search the list and see if we can find the search value
list_len = len(animals)

# search the loop - watch the indentation closely!
while i < list_len and found == False:
    if search_val.upper() == animals[i].upper():
        found = True
    # end if
    i = i + 1
# end while

# output - show the result of our search
if found == True:
    message = "Found " + search_val + " in the list.\n"
else:
    message = search_val + " was not found.\n"

print(message)

Parallel Array Search

#!/usr/bin/env python3

# Cuyamaca College CS-119
# In this demo we transition from arrays to lists as these are preferred
# in Python programming. Added bonus: They're easier!

# create a list of random animals and declare variables
animals = ["cat", "dog", "horse", "goat", "rabbit", "lizard", "duck"]
# create a parallel list of the qty of each animal available
qty_available = [15, 18, 0, 1, 12, 2, 1] 

search_val = ""
i = 0
found = False
message = ""
qty = 0

# input - find what we're looking for
search_val = input("Find what animal? ")

# process - search the list and see if we can find the search value
list_len = len(animals)

# watch indentation in the loop and selection structure closely!
while i < list_len and found == False:
    if search_val.upper() == animals[i].upper():
        # get qty from the parallel list qty_available
        qty = qty_available[i]
        found = True
    # end if
    i = i + 1
# end while

# output - show the result of our search
if found == True:
    message = "Found " + search_val + ". there are " str(qty) + " available\n"
else:
    message = search_val + " was not found.\n"

print(message)
